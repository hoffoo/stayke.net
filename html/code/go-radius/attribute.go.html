package radius

// Value constants used to represent the attribute's Value type
const (
    //1-253 octets containing UTF-8 encoded 10646 [7]
    //characters.  Text of length zero (0) MUST NOT be sent;
    //omit the entire attribute instead.
    Text byte = iota

    //1-253 octets containing binary data (values 0 through
    //255 decimal, inclusive).  Strings of length zero (0)
    //MUST NOT be sent; omit the entire attribute instead.
    String

    // 32 bit value, most significant octet first.
    Address

    //32 bit unsigned value, most significant octet first.
    Integer

    //32 bit unsigned value, most significant octet first --
    //seconds since 00:00:00 UTC, January 1, 1970.  The
    //standard Attributes do not use this data type but it is
    //presented here for possible use in future attributes.
    Time
)

type ByteAttr []byte
type Attr struct {
    ByteAttr

    Name     string
    VendorId uint16
    TypeId   uint8

    Value *[]byte
}

func NewAttr(v uint16, c uint8, t byte, name string) Attr {
    b := []byte{byte(c)}
    return Attr{b, name, v, c, nil}
}

func (a Attr) Code() int {
    return int(a.ByteAttr[0])
}

func (a Attr) Len() int {
    return len(a.ByteAttr)
}
